if( NOT __MULLE_OBJC_STANDARD_FOUNDATION_CMAKE_DEPENDENCIES_TXT__)
   set( __MULLE_OBJC_STANDARD_FOUNDATION_CMAKE_DEPENDENCIES_TXT__ ON)

   message( STATUS "# Include MulleObjCStandardFoundation CMakeDependencies.txt")

   include( ${DEPENDENCIES_DIR}/include/MulleObjC/CMakeDependencies.txt)
   include( ${DEPENDENCIES_DIR}/include/mulle_buffer/CMakeDependencies.txt)
   include( ${DEPENDENCIES_DIR}/include/mulle_container/CMakeDependencies.txt)
   include( ${DEPENDENCIES_DIR}/include/mulle_utf/CMakeDependencies.txt)
   include( ${DEPENDENCIES_DIR}/include/mulle_sprintf/CMakeDependencies.txt)

   #
   # for ppl including us
   #
   if( NOT MULLE_OBJC_STANDARD_FOUNDATION_LIBRARY)
      find_library( MULLE_OBJC_STANDARD_FOUNDATION_LIBRARY NAMES MulleObjCStandardFoundation)
      message(STATUS "MULLE_OBJC_STANDARD_FOUNDATION_LIBRARY is ${MULLE_OBJC_STANDARD_FOUNDATION_LIBRARY}")
      set( OBJC_DEPENDENCY_LIBRARIES
         ${MULLE_OBJC_STANDARD_FOUNDATION_LIBRARY}
         ${OBJC_DEPENDENCY_LIBRARIES}
         CACHE INTERNAL "need to cache this"
      )
      set( OBJC_DEPENDENCY_NAMES
         MulleObjCStandardFoundation
         ${OBJC_DEPENDENCY_NAMES}
         CACHE INTERNAL "need to cache this too"
      )
   endif()

   #
   # This library contains ___get_or_create_mulle_objc_universe and
   # the startup code to create the universe
   #
   if( NOT MULLE_OBJC_STANDARD_FOUNDATION_STARTUP_LIBRARY)
      find_library( MULLE_OBJC_STANDARD_FOUNDATION_STARTUP_LIBRARY NAMES MulleObjCStandardFoundationStartup)
      message( STATUS "MULLE_OBJC_STANDARD_FOUNDATION_STARTUP_LIBRARY is ${MULLE_OBJC_STANDARD_FOUNDATION_STARTUP_LIBRARY}")

      # overwrite what came before
      set( OBJC_STARTUP_LIBRARY
         ${MULLE_OBJC_STARTUP_LIBRARY}
         CACHE INTERNAL "need to cache this"
      )
   endif()


   if( MSVC)
      set( MULLE_OBJC_STANDARD_FOUNDATION_DEFINITIONS
         ${MULLE_OBJC_DEFINITIONS}
         ${MULLE_OBJC_BUFFER_DEFINITIONS}
         ${MULLE_OBJC_CONTAINER_DEFINITIONS}
         ${MULLE_OBJC_UTF_DEFINITIONS}
         ${MULLE_OBJC_SPRINTF_DEFINITIONS}
      )
      list( REMOVE_DUPLICATES MULLE_OBJC_STANDARD_FOUNDATION_DEFINITIONS)
   endif()
endif()
